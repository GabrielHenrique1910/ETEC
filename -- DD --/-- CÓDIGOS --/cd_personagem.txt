estado_idle.inicia = function()
{
    var _sprite = define_sprite(dir, spr_player_side, spr_player_back, spr_player_back);
    sprite_index = _sprite;
    image_index = 0;
}

estado_idle.roda = function()
{
    // Verificar se o jogador está se movendo
    if (cima || baixo || esquerda || direita)
    {
        troca_estado(estado_walk);
    }
}
#endregion

// Região estado_walk
#region estado_walk
estado_walk.inicia = function()
{
    dir = (point_direction(0, 0, direita - esquerda, baixo - cima) div 90) mod 4;
    sprite_index = define_sprite(dir, spr_plr_walk_s, spr_plr_walk_f, spr_plr_walk_b);
    image_index = 0;
}

estado_walk.roda = function()
{
    dir = (point_direction(0, 0, velh, velv) div 90) mod 4;

    if (velh != 0)
    {
        image_xscale = sign(velh);
    }
    else if (velv != 0)
    {
        image_xscale = sign(velv);
    }
    
    sprite_index = define_sprite(dir, spr_plr_walk_s, spr_plr_walk_f, spr_plr_walk_b);

    velv = (baixo - cima) * vel;
    velh = (direita - esquerda) * vel;

    // Trocar para o estado idle se não houver movimento
    if (velv == 0 && velh == 0)
    {
        troca_estado(estado_idle);
    }
}
#endregion
